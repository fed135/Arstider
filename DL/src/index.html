Arstider

Implementation

This is a quick showcase implementation of the class system, mixed with the brilliant require.js.

The code is also running right now, so you can check your console and do some tests.

For example, if you were to try "console.warn(DW.Main.getPrivate("bonjour"));" in the console, you would be surprised by the result.

var DW = {};

require(["arstider"],function(Ar){
	var Main = Ar.Class({
		requireOnce:true,
		name:"Main",
		body:function(){
			
			//===Class body goes here===//
			
			//Private
			var private_test = "pumpkin";
			var passcode = "bonjour";		//Let's just pretend here for a minute that the function and the rest of the javascript are 2 different itenties.
			
			//Public
			return {
				getPrivate:function(pass) {
					if(pass === passcode) {
						return "Passcode correct. Private was \""+private_test+"\""; //The rest of the program cannot read 'private_test' otherwise.
					}
				}
			};
			
			//===End of Class
		}()
	});
});

//Class is registered too... if you put it in a different file or call it later in the app.
require(["ar.Main"], function(Main){
	DW.Main = Main;
});
